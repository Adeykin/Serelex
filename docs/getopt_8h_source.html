00050 /*
00051  * getopt --
00052  * Parse argc/argv argument vector.
00053  */
00054 int
00055 getopt(int argc, char * const argv[], const char *optstring)
00056 {
00057    static char *place = EMSG;    /* option letter processing */
00058    char *oli;           /* option letter list index */
00059 
00060    if (optreset || *place == 0) {      /* update scanning pointer */
00061       optreset = 0;
00062       place = argv[optind];
00063       if (optind >= argc || *place++ != '-') {
00064          /* Argument is absent or is not an option */
00065          place = EMSG;
00066          return (-1);
00067       }
00068       optopt = *place++;
00069       if (optopt == '-' && *place == 0) {
00070          /* "--" => end of options */
00071          ++optind;
00072          place = EMSG;
00073          return (-1);
00074       }
00075       if (optopt == 0) {
00076          /* Solitary '-', treat as a '-' option
00077             if the program (eg su) is looking for it. */
00078          place = EMSG;
00079          if (strchr(optstring, '-') == NULL)
00080             return -1;
00081          optopt = '-';
00082       }
00083    } else
00084       optopt = *place++;
00085 
00086    /* See if option letter is one the caller wanted... */
00087    if (optopt == ':' || (oli = (char *)strchr(optstring, optopt)) == NULL) {
00088       if (*place == 0)
00089          ++optind;
00090       if (opterr && *optstring != ':')
00091          (void)fprintf(stderr,
00092                                       "unknown option -- %cn", optopt);
00093       return (BADCH);
00094    }
00095 
00096    /* Does this option need an argument? */
00097    if (oli[1] != ':') {
00098       /* don't need argument */
00099       optarg = NULL;
00100       if (*place == 0)
00101          ++optind;
00102    } else {
00103       /* Option-argument is either the rest of this argument or the
00104          entire next argument. */
00105       if (*place)
00106          optarg = place;
00107       else if (argc > ++optind)
00108          optarg = argv[optind];
00109       else {
00110          /* option-argument absent */
00111          place = EMSG;
00112          if (*optstring == ':')
00113             return (BADARG);
00114          if (opterr)
00115             (void)fprintf(stderr,
00116                                         "option requires an argument -- %cn",
00117                                         optopt);
00118          return (BADCH);
00119       }
00120       place = EMSG;
00121       ++optind;
00122    }
00123    return (optopt);        /* return option letter */
00124 }
